	// CONDITIONS -- The contition(s) for the effect to properly activate / trigger.
	//					  stdcond = Standard conditions i.e. keywords, that usually have arguments.
	//					  cond = More specific conditions. Here, we need a tuple of
	//					  		1. Ownership: Who's card or field we are waiting on.
	//							2. Location	: The location of the card. This can be specified explicitly or a range of values.
	//							3. Specifier: A card type, name, substype, attribute, or state. Must be specific values to target.
	//							4. Modifs	: Modifications that the specifier goes through. This can be another value the specifier could be.
	// TIMING  -- The timing the effect has. T is triggered (happens whenever), A is activated (when triggered, player given choice to activate)
	//				  and P is passive (constantly applied OR applied immediatley). 
	// TARGETS	--	At this point, we know when the effect will activate and what the conditions are for it to activate/trigger.
						Targets are vital; we need to know what the effect will end up effecting.
						target: A tuple of 4:
							1. Amount: How many things are we targeting? Can be only 1, 2, 3, 4, or all.
							2. Ownership: Same as in conditions. Must still be specified.
							3. Specifier: Same as in conditions. Must still be specified.
							4. Location: Where this card could be located.
						manual: Essentially a list of locations that we will target.
	// ACTION -- The actual changes we will be making to the card or area (can include attribute-based or state-based).
	//				 action: A tuple containing:
	//				 	1. function: What kind of change we will be making.
	//					2. attribute, state or location: The specific attribute/state/location we will be applying the change to.
	// DURATION -- How long the change will be applied.
	// NEXTLINK -- Signifies '+'. If the previous effect goes off, this one will be acknowledged too.


	program		::=	{ comment } 'START' { effect }_1 'END'
	comment		::=	'#' any_char '#'
	any_char		::=	'A' | 'B' | ... | 'a' | 'b' | ...
	effect		::=	'&'ordered_number '(' ({ statement }_1 | 'NULL') ')'
	statement	::=	{ descriptor }_1
	descriptor	::=	condition timing target action duration { nextLink }

	condition	::=	'<' { (stdcond | cond) (',')}_1 '>'
	stdcond		::=	'STDCOND' '->' { 'PASSIVE' | 'ENTER' | 'REVEAL' | any_phase | 'BlOODCALL' | 'PAYN' '%'number | 'PAY(N)' '%'number | ATTACKS | ... (',')}_1
	cond			::=	'COND' '->[ ' { 'AMOUNT='amount 'OWNER='ownership 'LOCATION='location 'SPECIFIER='specifier 'CHANGETO='value (',')}_1 ' ]'
	ownership	::=	'YOU' | 'OPPONENT' | 'BOTHPLAYERS' | 'ANYPLAYER' | 'OWNER'
	location		::=	( 'OFF1' | 'OFF2' | 'OFF3' | ... | 'BASE1' | 'BASE2' | 'BASE3' | ... | 'GRAVE' | 'HAND' ) | ( 'RANGE' '%'args ) | 'THISROW' | 'THISCOLUMN' | 'ANY'
	specifier	::=	cardtype | cardsubtype | cardname | attr | state
	value			::=	number | state | 'NONZERO'
	cardtype		::=	('UNIT' | 'BASE' | 'INVENTION' | ...) (cardsubtype)
	cardsubtype	::=	('OFFENSE' | 'DEFENSE' | 'SUPPORT' | 'SPECIAL' | 'PRIMARY' | 'SECONDARY' | ...)

	timing		::=	'@T' | '@A' | '@P'

	target		::=	'?' { '(AMOUNT='amount 'OWNER='ownership 'SPECIFIER='{specifier}_1' 'LOCATION=' {location}_1 ')' | manual (',')}_1 '?'
	manual		::=	{ location }_1
	amount		::=	'ONE' | 'TWO' | 'ONEORTWO' | 'ALL'

	action		::=	'!' { function {location} {attr | state} (',') }_1 '!'
	function		::=	'CREATE' '%'name '%'cardtype '%'cardsubtype 'WITH' '[' effect ']' | 'FLIP' | 'INC' '%'number | 'DEC' '%'number | 'DESTROY' | 'SEARCH' '%'(cardtype | cardsubtype | name_of_card) | 'DRAW' '%'number | 'MOVETO' '%'location | 'REVEAL'
	attr			::=	unitattr | baseattr | ... 
	state			::=	unitstate | basestate | ... 

	duration		::=	'DUR' ('ONCE' | 'TURN' '%'number | 'FOREVER')

	nextLink		::=	'THEN' statement
